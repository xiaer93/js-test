function isPalindrome(word) {
  // �쳣����
  if(word === undefined) {
    throw new Error('Invalid argument')
  }
  // ������� && �������
  if(typeof word !== 'number'){
    return false

  for(var i = 0, j = tmpWord.length - 1; i < j; ++i, --j) {

}
